SQL

JQiue2026/6/26...еӨ§зәҰ 21 еҲҶй’ҹж•°жҚ®еә“SQLMySQLMariaDBPostgreSQLSQLite

SQL жҳҜдёҖз§ҚеЈ°жҳҺејҸзҡ„йўҶеҹҹзү№е®ҡиҜӯиЁҖпјҢдё“й—Ёз”ЁдәҺз®ЎзҗҶж•°жҚ®еә“гҖӮдёҺ CгҖҒJava зӯүе‘Ҫд»ӨејҸиҜӯиЁҖдёҚеҗҢпјҢSQL дҫ§йҮҚдәҺе®ҡд№үжүҖйңҖз»“жһңпјҢиҖҢйқһе®һзҺ°жӯҘйӘӨгҖӮе…¶еҠҹиғҪж¶өзӣ–ж•°жҚ®жҹҘиҜўгҖҒдҝ®ж”№гҖҒз»“жһ„е®ҡд№үе’Ңи®ҝй—®жҺ§еҲ¶

SQL жҳҜе…ізі»еһӢж•°жҚ®еә“з®ЎзҗҶзі»з»ҹпјҲRDBMSпјүе…ұеҗҢйҒөеҫӘзҡ„иЎҢдёҡж ҮеҮҶгҖӮз„¶иҖҢпјҢдёҚеҗҢзҡ„ RDBMS дјҡеңЁж ҮеҮҶеҹәзЎҖдёҠеҸ‘еұ•еҮәиҮӘе·ұзҡ„SQL ж–№иЁҖпјҢиҖҢйқһе…ізі»еһӢж•°жҚ®еә“еҲҷйҖҡеёёйҮҮз”Ёе…¶д»–жҹҘиҜўиҜӯиЁҖ

SQL жңүйқһеёёйҮҚиҰҒзҡ„дёӨдёӘж ҮеҮҶпјҢеҲҶеҲ«жҳҜ SQL92 е’Ң SQL99пјҢеҲҶеҲ«д»ЈиЎЁдәҶ 92 е№ҙе’Ң 99 е№ҙйўҒеёғзҡ„ж ҮеҮҶпјҢеҰӮд»ҠдҪҝз”Ёзҡ„ SQL д»Қз„¶йҒөе®ҲзқҖиҝҷдәӣж ҮеҮҶпјҢSQL жҳҜдёҖз§ҚйҖҡз”ЁжҖ§ејәпјҢеҸҳеҢ–зӣёеҜ№е°‘пјҢдёҠжүӢе®№жҳ“зҡ„иҜӯиЁҖ

SQL е№¶дёҚеғҸе…¶д»–иҜӯиЁҖйӮЈж ·пјҢеӯҰд№ йңҖиҰҒеӨ§йҮҸзҡ„зЁӢеәҸеҹәзЎҖпјҢе®ғжӣҙеғҸжҳҜдёҖй—ЁиӢұиҜӯпјҢе°ұеҘҪеғҸз”ЁиӢұиҜӯеҚ•иҜҚеҗ‘ж•°жҚ®еә“иҝӣиЎҢеҜ№иҜқдёҖж ·пјҢжүҖд»Ҙ SQL иҜӯиЁҖиў«еҲҶдёәеӣӣдёӘйғЁеҲҶпјҡ

  • DDLпјҲData Definition Languageпјүж•°жҚ®е®ҡд№үиҜӯиЁҖпјҢз”ЁжқҘе®ҡд№үж•°жҚ®еә“еҜ№иұЎпјҢеҢ…жӢ¬ж•°жҚ®еә“гҖҒж•°жҚ®иЎЁе’ҢеҲ—пјҢеҸҜд»ҘеҲӣе»әгҖҒеҲ йҷӨе’Ңдҝ®ж”№ж•°жҚ®еә“е’ҢиЎЁзҡ„з»“жһ„
  • DMLпјҲData Manipulation Languageпјүж•°жҚ®ж“ҚдҪңиҜӯиЁҖпјҢз”ЁжқҘж“ҚдҪңе’Ңж•°жҚ®зӣёе…ізҡ„и®°еҪ•пјҢжҜ”еҰӮеўһеҠ гҖҒеҲ йҷӨгҖҒдҝ®ж”№ж•°жҚ®иЎЁдёӯзҡ„и®°еҪ•
  • DCLпјҲData Control Languageпјүж•°жҚ®жҺ§еҲ¶иҜӯиЁҖпјҢз”ЁжқҘе®ҡд№үи®ҝй—®жқғйҷҗе’Ңе®үе…Ёзә§еҲ«
  • DQLпјҲData Query Languageпјүж•°жҚ®жҹҘиҜўиҜӯиЁҖпјҢз”ЁжқҘжҹҘиҜўжғіиҰҒзҡ„и®°еҪ•пјҢе®ғжҳҜжңҖйҮҚиҰҒзҡ„дёҖйғЁеҲҶпјҢеӨ§еӨҡж•°жғ…еҶөдёӢйғҪжҳҜеңЁе’ҢжҹҘиҜўжү“дәӨйҒ“

DDL

CREATEе…ій”®еӯ—з”ЁжқҘеҲӣе»әж•°жҚ®еә“е’ҢиЎЁ

-- еҲӣе»әж•°жҚ®еә“
CREATE DATABASE database_name;
-- еҲӣе»әж•°жҚ®еә“еҗҢж—¶жҢҮе®ҡзј–з Ғ
CREATE DATABASE database_name CHARACTER SET еӯ—з¬ҰйӣҶ;
-- еҰӮжһңдёҚеӯҳеңЁе°ұеҲӣе»ә
CREATE DATABASE IF NOT EXISTS database_name
-- еҲӣе»әиЎЁ
CREATE TABLE table_name (
  column_name1 зұ»еһӢ зәҰжқҹ,
  column_name2 зұ»еһӢ зәҰжқҹ
);
-- зӨәдҫӢ
CREATE TABLE student (
  id int,
  stu_name varchar(50),
  stu_age int,
  stu_score int
);

ALTERе…ій”®еӯ—з”ЁжқҘдҝ®ж”№ж•°жҚ®еә“е’ҢиЎЁзҡ„ж“ҚдҪң

-- дҝ®ж”№ж•°жҚ®еә“
ALTER DATABASE database_name CHARACTER SET еӯ—з¬ҰйӣҶ
-- ж·»еҠ дёҖеҲ—
ALTER TABLE table_name ADD column_name зұ»еһӢ;
-- еҲ йҷӨдёҖеҲ—
ALTER TABLE table_name DROP column_name;
-- дҝ®ж”№еӯ—ж®өзұ»еһӢ
ALTER TABLE table_name MODIFY column_name зұ»еһӢ;
-- дҝ®ж”№иЎЁеҗҚ
RENAME TABLE curent_table_name to new_table_name;
-- дҝ®ж”№еҲ—еҗҚ
ALTER TABLE table_name CHANGE current_column_name new_column_name зұ»еһӢ;
-- дҝ®ж”№еӯ—з¬ҰйӣҶ
ALTER TABLE table_name CHARACTER SET еӯ—з¬ҰйӣҶ;

жҳҫзӨәеҲӣе»әиЎЁзҡ„дҝЎжҒҜ

SHOW CREATE TABLE table_name;

жҸҗзӨә

иЎЁзҡ„еӯ—з¬ҰйӣҶж— йңҖзү№ж„Ҹи®ҫзҪ®пјҢеӣ дёәе®ғи·ҹеҲӣе»әж•°жҚ®еә“ж—¶зҡ„зј–з ҒзӣёеҗҢ

DROPе…ій”®еӯ—з”ЁжқҘеҲ йҷӨж•°жҚ®еә“е’ҢиЎЁзҡ„ж“ҚдҪң

-- еҲ йҷӨж•°жҚ®еә“
DROP DATABASE database_name;
-- еҲ йҷӨиЎЁ
DROP TABLE table_name;
-- еҲ йҷӨдёҖдёӘеӯ—ж®ө
ALTER TABLE table_name DROP column_name;

INSERTз”ЁжқҘеҗ‘иЎЁдёӯжҸ’е…Ҙж•°жҚ®

-- жҸ’е…ҘдёҖжқЎж•°жҚ®
INSERT INTO table_name (column1, column2, ...)
VALUES (value1, value2, ...);
-- жҸ’е…ҘеӨҡжқЎж•°жҚ®
INSERT INTO table_name (column1, column2, ...)
VALUES (value1, value2, ...),
(value1, value2, ...);
-- еҸҜд»ҘзңҒз•Ҙеӯ—ж®өеҗҚпјҲеҝ…йЎ»жҸ’е…ҘжүҖжңүзҡ„еӯ—ж®өзҡ„еҖјпјү
INSERT INTO table_name
VALUES (value1, value2, ...),

жҸҗзӨә

жҸ’е…Ҙж•°жҚ®ж—¶пјҢеҲ—еҗҚдёҺеҲ—еҖјзҡ„зұ»еһӢгҖҒдёӘж•°гҖҒйЎәеәҸиҰҒдёҖдёҖеҜ№еә”пјҢеҖјдёҚиғҪи¶…иҝҮеҲ—е®ҡд№үзҡ„й•ҝеәҰпјҢжҸ’е…Ҙеӯ—з¬ҰдёІе’Ңж—Ҙжңҹж—¶пјҢеҝ…йЎ»дҪҝз”ЁеҚ•еј•еҸ·

UPDATEз”ЁжқҘжӣҙж–°иЎЁдёӯзҡ„ж•°жҚ®

UPDATE table_name SET column1 = value1, column2 = value2, ...
-- ж №жҚ®жқЎд»¶иҜӯеҸҘдҝ®ж”№жҢҮе®ҡеҲ—зҡ„еҖј
UPDATE table_name SET column1 = value1, column2 = value2, ... where column = value

DELETEеҲ йҷӨи®°еҪ•

-- й»ҳи®ӨеҲ йҷӨжүҖжңүзҡ„и®°еҪ•
DELETE FROM table_name;
-- еҲ йҷӨжҹҗдёҖжқЎи®°еҪ•
DELETE FROM table_name where column = value;

TRUNCATEеҲ йҷӨжүҖжңүи®°еҪ•е№¶жҲӘж–ӯиЎЁзҡ„з»“жһ„

TRUNCATE TABLE table_name;

DELETE е’Ң TRUNCATE

DELETEеҸӘдјҡеҲ йҷӨи®°еҪ•пјҢдҪҶжҳҜиЎЁзҡ„з»“жһ„иҝҳеңЁ TRUNCATEеҲ йҷӨи®°еҪ•пјҢиЎЁгҖҒзҙўеј•гҖҒеӨ–й”®гҖҒжқғйҷҗе…ЁйғЁдҝқз•ҷпјҢе№¶дё”йҮҚзҪ® AUTO_INCREMENTпјҢйҖҹеәҰжҜ”DELETEеҝ«

DQL

DQL иҜӯеҸҘдёҚдјҡеҜ№ж•°жҚ®иҝӣиЎҢж”№еҸҳпјҢиҝҷдёӘж•°жҚ®дјҡд»Һж•°жҚ®еә“дёӯиҜ»еҲ°еҶ…еӯҳпјҢд»Ҙиҝҷз§ҚиЎЁзҡ„еҪўејҸжҹҘиҜўеҮәжқҘзҡ„ж•°жҚ®з§°дёәиҷҡжӢҹз»“жһңйӣҶпјҢSELECTжҳҜ SQL дёӯжңҖйҮҚиҰҒзҡ„е…ій”®еӯ—

иҝҷдёӘиЎЁдёәдёӢеҲ—жҹҘиҜўжҸҗдҫӣж”ҜжҢҒпјҡ

CREATE TABLE student (
  id int,
  stu_name varchar(50),
  stu_age int,
  stu_gender varchar(1),
  stu_score int
);

*жҹҘеҮәжүҖжңүеҲ—

SELECT * FROM student;

жҹҘиҜўжҢҮе®ҡеҲ—

SELECT stu_name, stu_age, ... FROM student;

дҪҝз”Ёasе°ҶжҹҘиҜўеҮәжқҘзҡ„еӯ—ж®өжҢҮе®ҡеҲ«еҗҚпјҢеҸҜд»ҘзңҒз•ҘдёҚеҶҷ

SELECT stu_name AS name, stu_age AS age, ... FROM student; 
SELECT stu_name name, stu_age age, ... FROM student;

еӯ—ж®өжҺ§еҲ¶

дҪҝз”ЁDISTINCTиғҪеӨҹеҜ№жҹҘиҜўзҡ„еҲ—иҝӣиЎҢеҺ»йҮҚ

SELECT DISTINCT stu_name FROM student;

еҰӮжһңжҹҘиҜўзҡ„еҲ—жҳҜж•°еӯ—зұ»еһӢеҸҜд»Ҙиҝ”еӣһи®Ўз®—еҗҺзҡ„з»“жһң

-- еҲ—зҡ„з»“жһңеҸҜд»ҘеҒҡиҝҗз®—еҫ—еҲ°еҪўжҲҗж–°зҡ„еҲ—
SELECT age + 10 from student
-- е’Ңе…¶д»–еҲ—иҝӣиЎҢи®Ўз®—
SELECT math + english from table_name
SELECT stu_age, stu_score, stu_score + stu_age FROM student;

йҖҡиҝҮIFNULL()еҮҪж•°еӨ„зҗҶдёәеҲ—дёәз©әзҡ„еҖјпјҢеҰӮжһңдёәз©әпјҢеҲҷдјҡиў«и§Ҷдёәз¬¬дәҢдёӘеҸӮж•°

SELECT stu_age, stu_score, IFNULL(stu_score, 0) + IFNULL(stu_age, 0) FROM student;

жқЎд»¶жҹҘиҜў

жқЎд»¶жҹҘиҜўе°ұжҳҜж №жҚ®whereе…ій”®еӯ—жҢҮе®ҡзҡ„жқЎд»¶иҝӣиЎҢжҹҘиҜўпјҢз¬ҰеҗҲжқЎд»¶зҡ„жүҚдјҡиў«жҹҘиҜўеҮәжқҘ

SELECT * FROM table_name where column = value;

иҝҗз®—з¬ҰдёҚжӯўжңү=пјҢиҝҳжңүдёӢеҲ—еҸҜз”Ёзҡ„иҝҗз®—з¬Ұпјҡ

иҝҗз®—з¬ҰжҸҸиҝ°дҫӢеӯҗи§ЈйҮҠ
=зӯүдәҺSELECT * FROM student WHERE stu_name = 'zs'жҹҘиҜў stu_name зӯүдәҺ 'zs' зҡ„ж•°жҚ®
!=дёҚзӯүдәҺSELECT * FROM student WHERE stu_name != 'zs'жҹҘиҜў stu_name дёҚзӯүдәҺ 'zs' зҡ„ж•°жҚ®
<>дёҚзӯүдәҺSELECT * FROM student WHERE stu_name <> 'zs'жҹҘиҜў stu_name дёҚзӯүдәҺ 'zs' зҡ„ж•°жҚ®
<е°ҸдәҺSELECT * FROM student WHERE stu_age < 18жҹҘиҜў stu_age е°ҸдәҺ 18 зҡ„ж•°жҚ®
>еӨ§дәҺSELECT * FROM student WHERE stu_age > 18жҹҘиҜў stu_age еӨ§дәҺ 18 зҡ„ж•°жҚ®
<=е°ҸдәҺзӯүдәҺSELECT * FROM student WHERE stu_age <= 18жҹҘиҜў stu_age е°ҸдәҺзӯүдәҺ 18 зҡ„ж•°жҚ®
>=еӨ§дәҺзӯүдәҺSELECT * FROM student WHERE stu_age > 18жҹҘиҜў stu_age еӨ§дәҺзӯүдәҺ 18 зҡ„ж•°жҚ®
BETWEEN...AND...еҖјеңЁд»Җд№ҲиҢғеӣҙSELECT * FROM student WHERE id BETWEEN 10001 AND 10003жҹҘиҜў id еҖј 10001 еҲ° 10003 д№Ӣй—ҙзҡ„ж•°жҚ®
IN(set)еӣәе®ҡиҢғеӣҙеҖјSELECT * FROM student WHERE id IN (10001, 10002, 10003)жҹҘиҜў id дёә10001,10002,10003зҡ„еҖј
IS NULLдёәз©әSELECT * FROM student WHERE stu_age IS NULLжҹҘиҜў stu_age дёә NULL зҡ„еӯҰз”ҹ
ANDдёҺSELECT * FROM student WHERE id = 10001 AND stu_name = 'zs'жҹҘиҜў id дёә 10001 дё” stu_name дёә 'zs' зҡ„ж•°жҚ®
ORжҲ–SELECT * FROM student WHERE id = 10001 OR stu_name = 'zs'жҹҘиҜў id дёә 10001 жҲ– stu_name дёә 'zs' зҡ„ж•°жҚ®
NOTйқһSELECT * FROM student WHERE stu_gender IS NOT NULLжҹҘиҜў stu_gender дёҚдёә NULL зҡ„ж•°жҚ®

жЁЎзіҠжҹҘиҜў

йҖҡиҝҮдёҖз§ҚзӣёеҜ№жқЎд»¶жҜ”иҫғжЁЎзіҠзҡ„ж–№ејҸжқҘжҹҘиҜўж•°жҚ®пјҢдҪҝз”ЁLIKEе…ій”®еӯ—жқҘжһ„жҲҗжқЎд»¶пјҢеҸҲеңЁе…¶жқЎд»¶дёӯжҸҗдҫӣдәҶеҮ дёӘйҖҡй…Қз¬ҰжқҘй…ҚеҗҲпјҡ

  • _пјҡд»»ж„ҸдёҖдёӘеӯ—з¬Ұ
  • %пјҡ0 ~ n дёӘеӯ—з¬Ұ
-- жҹҘиҜўд»»ж„Ҹ 2 дёӘеӯ—з¬Ұ
SELECT * FROM student WHERE stu_name LIKE '__';
-- жҹҘиҜўд»Ҙ s з»“е°ҫзҡ„ 2 дёӘеӯ—з¬Ұ
SELECT * FROM student WHERE stu_name LIKE '_s';
-- жҹҘиҜўд»Ҙ z ејҖеӨҙзҡ„ 2 дёӘеӯ—з¬Ұ
SELECT * FROM student WHERE stu_name LIKE 'z_';
-- жҹҘиҜўд»Ҙ z ејҖеӨҙзҡ„ n дёӘеӯ—з¬Ұ
SELECT * FROM student WHERE stu_name LIKE 'z%';
-- жҹҘиҜўд»Ҙд»»ж„Ҹеӯ—з¬ҰејҖеӨҙдё”з¬¬дәҢдёӘеӯ—з¬Ұдёә u зҡ„д»»ж„ҸдёӘеӯ—з¬Ұ
SELECT * FROM student WHERE stu_name LIKE '_u%';
-- жҹҘиҜўеҢ…еҗ« s зҡ„еӯ—з¬Ұ
SELECT * FROM student WHERE stu_name LIKE '%s%';

жҺ’еәҸ

й»ҳи®Өжғ…еҶөдёӢпјҢж•°жҚ®зҡ„жҺ’еәҸйЎәеәҸжҳҜдёҚзЎ®е®ҡзҡ„гҖӮиҝҷж„Ҹе‘ізқҖжҹҘиҜўз»“жһңзҡ„йЎәеәҸеҸҜиғҪдјҡйҡҸзқҖж—¶й—ҙгҖҒж•°жҚ®жҸ’е…ҘжҲ–е…¶д»–еӣ зҙ иҖҢеҸҳеҢ–пјҢMySQL зҡ„й»ҳи®ӨжҺ’еәҸи§„еҲҷжҳҜеҹәдәҺж•°жҚ®еӯҳеӮЁзҡ„зү©зҗҶйЎәеәҸгҖӮеҪ“ж•°жҚ®жҸ’е…ҘеҲ°иЎЁдёӯж—¶пјҢе®ғд»¬йҖҡеёёжҢүз…§е®ғд»¬еңЁзЈҒзӣҳдёҠзҡ„еӯҳеӮЁйЎәеәҸиҝӣиЎҢжҺ’еәҸгҖӮз„¶иҖҢпјҢиҝҷз§Қзү©зҗҶжҺ’еәҸе№¶дёҚжҳҜеҸҜйқ зҡ„пјҢORDER BYе…ій”®еӯ—еҸҜд»ҘеҜ№з»“жһңиҝӣиЎҢеҸҜйқ жҺ’еәҸ

-- ж №жҚ® stu_score иҝӣиЎҢеҚҮеәҸпјҢй»ҳи®Өе…ій”®еӯ—`ASC`еҸҜд»ҘзңҒз•ҘдёҚеҶҷ
SELECT * FROM student ORDER BY stu_score ASC;
-- ж №жҚ® stu_score иҝӣиЎҢйҷҚеәҸпјҢж·»еҠ е…ій”®еӯ—`DESC`
SELECT * FROM student ORDER BY stu_score DESC;

иҝҳеҸҜд»ҘиҝӣиЎҢз»„еҗҲжҺ’еәҸ

-- жҹҘиҜў stu_score жҢүйҷҚеәҸжҺ’еҲ—пјҢеҰӮжһң stu_score зӣёеҗҢе°ұжҢүеҚҮеәҸжҺ’еҲ—
SELECT * FROM student ORDER BY stu_score DESC, stu_age ASC;

SELECT stu_gender, GROUP_CONCAT(stu_score) FROM student WHERE stu_score >= 60 GROUP BY stu_gender HAVING SUM(stu_score) > 190;

еңЁж•°жҚ®еә“зі»з»ҹдјҡжңүеҚ•зӢ¬зҡ„еӯҳеӮЁз©әй—ҙеӯҳж”ҫвҫ®иҒҡйӣҶзҙўеј•пјҢиҝҷдәӣзҙўеј•йЎ№жҳҜжҢүз…§йЎәеәҸеӯҳеӮЁзҡ„пјҢдҪҶзҙўеј•йЎ№жҢҮеҗ‘зҡ„еҶ…е®№жҳҜйҡҸжңәеӯҳеӮЁзҡ„гҖӮд№ҹе°ұжҳҜиҜҙзі»з»ҹдјҡиҝӣвҫҸдёӨж¬ЎжҹҘжүҫпјҢз¬¬вјҖж¬Ўе…ҲжүҫеҲ°зҙўеј•пјҢз¬¬вјҶж¬ЎжүҫеҲ°зҙўеј•еҜ№еә”зҡ„дҪҚзҪ®еҸ–еҮәж•°жҚ®вҫҸгҖӮвҫ®иҒҡйӣҶзҙўеј•дёҚдјҡжҠҠзҙўеј•жҢҮеҗ‘зҡ„еҶ…е®№еғҸиҒҡйӣҶзҙўеј•вјҖж ·зӣҙжҺҘж”ҫеҲ°зҙўеј•зҡ„еҗҺвҫҜпјҢвҪҪжҳҜз»ҙжҠӨеҚ•зӢ¬зҡ„зҙўеј•иЎЁпјҲеҸӘз»ҙжҠӨзҙўеј•пјҢдёҚз»ҙжҠӨзҙўеј•жҢҮеҗ‘зҡ„ж•°жҚ®пјүпјҢдёәж•°жҚ®жЈҖзҙўжҸҗдҫӣвҪ…дҫҝгҖӮжҲ‘д»¬иҝҳд»ҘжұүиҜӯеӯ—е…ёдёәдҫӢпјҢеҰӮжһңжғіиҰҒжҹҘжүҫвҖңж•°вҖқеӯ—пјҢйӮЈд№ҲжҢүз…§йғЁвҫёжҹҘжүҫзҡ„вҪ…ејҸпјҢе…ҲжүҫеҲ°вҖңж•°вҖқеӯ—зҡ„еҒҸж—ҒйғЁвҫёпјҢз„¶еҗҺиҝҷдёӘвҪ¬еҪ•дјҡе‘ҠиҜүжҲ‘д»¬вҖңж•°вҖқеӯ—еӯҳж”ҫеҲ°з¬¬еӨҡе°‘в»ҡпјҢеҶҚеҺ»жҢҮе®ҡзҡ„в»ҡз ҒжүҫиҝҷдёӘеӯ—
